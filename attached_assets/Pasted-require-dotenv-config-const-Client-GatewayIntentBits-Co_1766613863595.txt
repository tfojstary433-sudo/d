require("dotenv").config();

const { Client, GatewayIntentBits, Collection } = require("discord.js");
const fs = require("fs");
const path = require("path");

// ğŸ” Debug tokena (opcjonalne â€“ usuÅ„ po sprawdzeniu)
console.log("DISCORD_TOKEN:", process.env.DISCORD_TOKEN);
console.log("TOKEN LENGTH:", process.env.DISCORD_TOKEN?.length);

// ğŸ¤– Klient Discorda
const client = new Client({
    intents: [GatewayIntentBits.Guilds],
});

client.commands = new Collection();

// ğŸ“‚ Åadowanie komend
const commandsPath = path.join(__dirname, "commands");

if (!fs.existsSync(commandsPath)) {
    console.error("âŒ Folder 'commands' nie istnieje!");
    process.exit(1);
}

const commandFiles = fs
    .readdirSync(commandsPath)
    .filter(file => file.endsWith(".js"));

for (const file of commandFiles) {
    const filePath = path.join(commandsPath, file);
    const command = require(filePath);

    if (!command.data || !command.execute) {
        console.error(`âŒ BÅÄ„D: ${file} nie posiada 'data' lub 'execute'`);
        continue;
    }

    client.commands.set(command.data.name, command);
    console.log(`âœ… ZaÅ‚adowano komendÄ™: ${command.data.name}`);
}

// âš¡ ObsÅ‚uga Slash Commands
client.on("interactionCreate", async interaction => {
    if (!interaction.isChatInputCommand()) return;

    const command = client.commands.get(interaction.commandName);
    if (!command) {
        console.warn(`âš ï¸ Nie znaleziono komendy: ${interaction.commandName}`);
        return;
    }

    try {
        await command.execute(interaction);
    } catch (error) {
        console.error(error);

        const errorMessage = {
            content: "âŒ WystÄ…piÅ‚ bÅ‚Ä…d podczas wykonywania komendy!",
            ephemeral: true,
        };

        if (interaction.replied || interaction.deferred) {
            await interaction.followUp(errorMessage);
        } else {
            await interaction.reply(errorMessage);
        }
    }
});

// âœ… GotowoÅ›Ä‡ bota
client.once("ready", () => {
    console.log(`ğŸ¤– Zalogowano jako ${client.user.tag}`);
});

// ğŸš€ Logowanie
if (!process.env.DISCORD_TOKEN) {
    console.error("âŒ Brak DISCORD_TOKEN w pliku .env");
    process.exit(1);
}

client.login(process.env.DISCORD_TOKEN);
